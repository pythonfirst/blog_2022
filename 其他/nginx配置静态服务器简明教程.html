    云服务器配置nginx静态站点

    zhaocanxiang 2022/4/17

    # 源起

    最近心血来潮，折腾一个 vuepress 博客部署到腾讯云上。因为之前折腾过一次 hexo 的博客，以为这次肯定很轻松就能搞定。结果在 nginx 配置上还是遇到一个小小的坑，这篇文章用来记录一下。

    云服务器的登录配置这里不再赘述，网上有很多文章讲，直接从 nginx 配置开始。

    # 安装 nginx

    sudo apt update
sudo apt nginx;

    # 启动 nginx

    # 启用 Nginx
systemctl start nginx

# 设置为在系统启动时自动启动
systemctl enable nginx

    # 修改配置文件

    # 含有 sites-enable/default 默认文件

    不同版本的 nginx 这里会有差别，主要体现在 nginx 配置文件上。我安装的版本配置文件存在两个分别为 etc/nginx/nginx.conf 和 etc/nginx/sites-enabled/default。相当于 nginx 会将两个文件组合在一起。此时起一个最简单的静态文件服务器，需要修改 default 文件。

    default 文件已经配置了默认监听端口 80, 因为只是测试静态服务，我们只改一下 root 指令配置即可: root /home/www/html;

    server {
        listen 80 default_server;
        listen [::]:80 default_server;å

        # 主要配置80端口监听的根目录
        root /home/www/html;

        # Add index.php to the list if you are using PHP
        index index.html index.htm index.nginx-debian.html;

        server_name _;

        location / {
                # First attempt to serve request as file, then
                # as directory, then fall back to displaying a 404.
                try_files $uri $uri/ =404;
        }
}

    # 普通的 nginx.conf 配置文件

    配置的 root 同上，只不过因为不需要合并 default，需要在 etc/nginx/nginx.conf 文件中的 http 块添加 server 块。

    http {
 server {
        listen 80 default_server;
        listen [::]:80 default_server;å

        # 主要配置80端口监听的根目录
        root /home/www/html;

        # Add index.php to the list if you are using PHP
        index index.html index.htm index.nginx-debian.html;

        server_name _;

        location / {
                # First attempt to serve request as file, then
                # as directory, then fall back to displaying a 404.
                try_files $uri $uri/ =404;
        }
 }
}

    # 创建测试文件

    # 创建静态文件目录
mkdir -p home/www/html;

cd home/www/html

# 创建index.html
touch index.html;

vim index.html  # 随便写一些内容即可。

    # 重启 nginx

    # 重新启动 nginx，以使更改的配置生效。
nginx -s reload
